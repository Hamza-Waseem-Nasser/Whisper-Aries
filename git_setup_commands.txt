# Git Setup Commands for Whisper-Aries Project
# Run these commands after installing Git for Windows

# 1. Initialize git repository
git init

# 2. Add remote repository
git remote add origin https://github.com/Hamza-Waseem-Nasser/Whisper-Aries.git

# 3. Configure git (replace with your details)
git config user.name "Your Name"
git config user.email "your.email@example.com"

# 4. Add all files to staging
git add .

# 5. Make initial commit
git commit -m "Initial commit: Whisper Large-v3 transcription system

- Production-ready Arabic/English transcription setup
- GPU acceleration with CUDA 12.1 support
- INT8 quantization for faster processing
- Automated installation script for team onboarding
- Comprehensive documentation and troubleshooting
- Multiple output formats (JSON, TXT, SRT)
- Batch processing capabilities"

# 6. Push to GitHub
git branch -M main
git push -u origin main
